439 Pt3d NodalFuncImpl::ComputeGradientForPoint(int a_ptIdx, const std::vector<InterpPtInfo>& a_pts)
440 {
441  Pt3d grad;
442  if (m_2d)
443  {
444  double A[9] = {0, 0, 0, 0, 0, 0, 0, 0, 0};
445  NodalFunc2d(a_ptIdx, a_pts, A);
446  grad.x = A[0];
447  grad.y = A[1];
448  return grad;
449  }
450 
451  // 3D
452  double A[3][3], x[3], B[3], *Aptr[3];
453  for (int i = 0; i < 3; ++i)
454  {
455  Aptr[i] = &A[i][0];
456  x[i] = B[i] = 0;
457  for (int j = 0; j < 3; ++j)
458  A[i][j] = 0;
459  }
460 
461  const Pt3d& p = m_pts[a_ptIdx];
462  double dx, dy, dz(0), df, weight;
463  for (size_t i = 0; i < a_pts.size(); ++i)
464  {
465  weight = a_pts[i].m_weight;
466  dx = a_pts[i].m_loc.x - p.x;
467  dy = a_pts[i].m_loc.y - p.y;
468  if (!m_2d)
469  dz = a_pts[i].m_loc.z - p.z;
470  df = a_pts[i].m_scalar - m_scalar[a_ptIdx];
471 
472  A[0][0] += weight * dx * dx;
473  A[0][1] += weight * dx * dy;
474  A[0][2] += weight * dx * dz;
475  A[1][1] += weight * dy * dy;
476  A[1][2] += weight * dy * dz;
477  A[2][2] += weight * dz * dz;
478 
479  B[0] += weight * df * dx;
480  B[1] += weight * df * dy;
481  B[2] += weight * df * dz;
482  }
483  A[1][0] = A[0][1];
484  A[2][0] = A[0][2];
485  A[2][1] = A[1][2];
486 
487  bool err = false;
488  if (m_2d)
489  err = !mxSolveNxN(&Aptr[0], x, B, 2);
490  else
491  err = !mxSolve3x3(A, x, B);
492  if (err || m_debugTest)
493  {
494  if (B[0] == 0.0 && B[1] == 0.0 && B[2] == 0.0)
495  { // ds was probably zero which is OK. Apply the zero or trivial solution
496  x[0] = x[1] = x[2] = 0.0;
497  }
498  else if (!m_errorReport)
499  {
500  boost::mutex mtx;
501  mtx.lock();
502  m_errorReport = true;
503  std::string msg =
504  "Error computing nodal functions. Increase the number "
505  "of points used for nodal function calculation.";
506  XM_LOG(xmlog::error, msg);
507  mtx.unlock();
508  }
509  }
510  grad.x = x[0];
511  grad.y = x[1];
512  grad.z = x[2];
513  return grad;
514 } // NodalFuncImpl::ComputeGradientForPoint

523 double NodalFuncImpl::ScalarFromPtIdx(int a_ptIdx, const Pt3d& a_loc) const
524 {
525  if (1 == m_type)
526  {
527  if (m_2d)
528  {
529  double dc, qi;
530  dc = m_gradient[a_ptIdx].x * m_pts[a_ptIdx].x + m_gradient[a_ptIdx].y * m_pts[a_ptIdx].y -
531  m_scalar[a_ptIdx];
532  qi = m_gradient[a_ptIdx].x * a_loc.x + m_gradient[a_ptIdx].y * a_loc.y - dc;
533  return qi;
534  }
535 
536  double f;
537  f = m_scalar[a_ptIdx] + m_gradient[a_ptIdx].x * (a_loc.x - m_pts[a_ptIdx].x) +
538  m_gradient[a_ptIdx].y * (a_loc.y - m_pts[a_ptIdx].y) +
539  m_gradient[a_ptIdx].z * (a_loc.z - m_pts[a_ptIdx].z);
540  return f;
541  }
542  else
543  {
544  double fi = (double)m_scalar[a_ptIdx];
545  double qi, dx, dy, dz;
546  dx = a_loc.x - m_pts[a_ptIdx].x;
547  dy = a_loc.y - m_pts[a_ptIdx].y;
548  if (m_2d)
549  {
550  qi = fi + m_quadratic[a_ptIdx][0] * dx + m_quadratic[a_ptIdx][1] * dy +
551  m_quadratic[a_ptIdx][2] * dx * dx + m_quadratic[a_ptIdx][3] * dx * dy +
552  m_quadratic[a_ptIdx][4] * dy * dy;
553  return qi;
554  }
555 
556  dz = a_loc.z - m_pts[a_ptIdx].z;
557  qi = fi + m_quadratic[a_ptIdx][0] * dx + m_quadratic[a_ptIdx][1] * dy +
558  m_quadratic[a_ptIdx][2] * dz + m_quadratic[a_ptIdx][3] * dx * dy +
559  m_quadratic[a_ptIdx][4] * dx * dz + m_quadratic[a_ptIdx][5] * dy * dz +
560  m_quadratic[a_ptIdx][6] * dx * dx + m_quadratic[a_ptIdx][7] * dy * dy +
561  m_quadratic[a_ptIdx][8] * dz * dz;
562  return qi;
563  }
