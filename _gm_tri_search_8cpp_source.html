275 {
276  if (m_pts->empty() || m_tris->empty())
277  return;
278 
279  m_min = XM_FLT_HIGHEST;
280  m_max = XM_FLT_LOWEST;
281  for (size_t i = 0; i < m_pts->size(); ++i)
282  {
283  Pt3d& pt = (*m_pts)[i];
284  if (pt.x < m_min.x)
285  m_min.x = pt.x;
286  if (pt.y < m_min.y)
287  m_min.y = pt.y;
288  if (pt.z < m_min.z)
289  m_min.z = pt.z;
290  if (pt.x > m_max.x)
291  m_max.x = pt.x;
292  if (pt.y > m_max.y)
293  m_max.y = pt.y;
294  if (pt.z > m_max.z)
295  m_max.z = pt.z;
296  }
297  double tol = sqr(m_max.x - m_min.x) + sqr(m_max.y - m_min.y);
298  tol = sqrt(tol) / 1e9;
299 
300  qRtree rtree;
301  size_t nTri = m_tris->size() / 3;
302  idx_tri getter(&(*m_pts)[0], &(*m_tris)[0], tol);
303  m_rTree = new bgi::rtree<size_t, qRtree, idx_tri>(
304  boost::counting_iterator<size_t>(0), boost::counting_iterator<size_t>(nTri), rtree, getter);

481 int GmTriSearchImpl::FindTriangle(const Pt3d& a_pt, int ix[3], Pt3d& weights)
482 {
483  if (!m_tris || m_tris->empty() || !m_pts || m_pts->empty())
484  {
485  XM_LOG(xmlog::error, "Unable to find triangle; no points or triangles exist.");
486  return XM_NONE;
487  }
488  Pt3d p(a_pt.x, a_pt.y, 0);
489  std::vector<value> vals;
490  m_rTree->query(bgi::intersects(p), std::back_inserter(vals));
491  if (vals.empty())
492  return XM_NONE;
493 
494  size_t idx;
495  for (size_t i = 0; i < vals.size(); ++i)
496  {
497  idx = vals[i] * 3;
498  // if any point is inactive then skip this triangle
499  if (!ActiveTri((int)idx))
500  continue;
501  ix[0] = (*m_tris)[idx];
502  ix[1] = (*m_tris)[idx + 1];
503  ix[2] = (*m_tris)[idx + 2];
504 
505  GmBstPoly3d poly;
506  for (size_t j = 0; j < 3; ++j)
507  {
508  bg::exterior_ring(poly).push_back((*m_pts)[ix[j]]);
509  }
510  bg::exterior_ring(poly).push_back((*m_pts)[ix[0]]);
511  if (bg::covered_by(a_pt, poly))
512  {
513  BarycentricVals b;
514  GetTriBarycentricVals((int)idx, ix, b);
515  iGetBarycentricCoords(p, b, weights);
516  return (int)idx;
517  }
518  }
519  // additional checks with some tolerancing
520  for (size_t i = 0; i < vals.size(); ++i)
521  {
522  idx = vals[i] * 3;
523  // if any point is inactive then skip this triangle
524  if (!ActiveTri((int)idx))
525  continue;
526  ix[0] = (*m_tris)[idx];
527  ix[1] = (*m_tris)[idx + 1];
528  ix[2] = (*m_tris)[idx + 2];
529  BarycentricVals b;
530  GetTriBarycentricVals((int)idx, ix, b);
531  iGetBarycentricCoords(p, b, weights);
532  // if (GTEQ_TOL(weights.x, 0, BARYTOL) && GTEQ_TOL(weights.y, 0, BARYTOL) &&
533  // GTEQ_TOL(weights.z, 0, BARYTOL)) {
534  if (weights.x >= -BARYTOL && weights.y >= -BARYTOL && weights.z >= -BARYTOL)
535  {
536  return (int)idx;
537  }
538  }
539 
540  return XM_NONE;
541 } // GmTriSearchImpl::FindTriangle
