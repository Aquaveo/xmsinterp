110 void TrOuterTriangleDeleterImpl::Delete(const VecInt2d& a_polys, BSHP<TrTin> a_tin)
111 {
112  m_polys = a_polys;
113  m_tin = a_tin;
114 
115  // Make sure polygons are closed
116  for (int i = 0; i < (int)m_polys.size(); ++i)
117  {
118  if (m_polys[i].front() != m_polys[i].back())
119  {
120  XM_ASSERT(false);
121  m_polys[i].push_back(m_polys[i].front());
122  }
123  }
124 
125  // Flag triangles on border
126  VecInt flags(m_tin->NumTriangles(), BE_UNVISITED);
127  for (size_t i = 0; i < m_polys.size(); ++i)
128  {
129  FlagTrianglesAlongPolygon(m_polys[i], flags);
130  }
131  if (m_observer)
132  m_observer->ProgressStatus(0.4);
133 
134  // Flag all triangles working out from the ones already flagged
135  MarkNeighbors(flags);
136  if (m_observer)
137  m_observer->ProgressStatus(0.8);
138 
139  // Create the set of triangles to delete
140  SetInt trisToDelete;
141  for (size_t i = 0; i < flags.size(); ++i)
142  {
143  if (flags[i] & BE_OUTSIDE)
144  {
145  trisToDelete.insert((int)i);
146  }
147  }
148 
149  m_tin->DeleteTriangles(trisToDelete);
150  if (m_observer)
151  m_observer->ProgressStatus(1.0);
152 } // TrOuterTriangleDeleterImpl::Delete

158 void TrOuterTriangleDeleterImpl::FlagTrianglesAlongPolygon(const VecInt& a_poly, VecInt& a_flags)
159 {
160  VecInt::const_iterator it0 = a_poly.begin();
161  VecInt::const_iterator it1 = it0;
162  ++it1;
163  int outTri, inTri;
164  for (; it1 != a_poly.end(); ++it0, ++it1)
165  {
166  // outTri = m_tin->TriangleAdjacentToEdge(*it0, *it1);
167  // inTri = m_tin->TriangleAdjacentToEdge(*it1, *it0);
168  inTri = m_tin->TriangleAdjacentToEdge(*it0, *it1);
169  outTri = m_tin->TriangleAdjacentToEdge(*it1, *it0);
170  if (outTri != -1)
171  {
172  // A polygon with an internal arc is a little different. We have adjacent
173  // triangles to 2 edges that will be marked as "inside".
174  if (!(a_flags[outTri] & BE_INSIDE))
175  a_flags[outTri] = BE_OUTSIDE | BE_ONBOUNDARY;
176  }
177  if (inTri != -1)
178  {
179  a_flags[inTri] = BE_INSIDE | BE_ONBOUNDARY;
180  }
181  }

190 void TrOuterTriangleDeleterImpl::MarkNeighbors(VecInt& a_flags)
191 {
192  XM_ENSURE_TRUE_VOID_NO_ASSERT(!m_tin->Triangles().empty());
193 
194  // Create stack size of num triangles
195  int numTri = m_tin->NumTriangles();
196  VecInt stack(numTri, -1);
197 
198  bool done = false;
199  int lastTri = 0;
200  while (!done)
201  {
202  // find a visited elem with unvisited neighbors
203  int tri = lastTri;
204  bool found = false;
205  while (tri < numTri && tri != -1 && !found)
206  {
207  if (a_flags[tri] & BE_ONBOUNDARY && !(a_flags[tri] & BE_VISITED))
208  {
209  for (int id = 0; id < 3 && !found; ++id)
210  {
211  int adjTri = m_tin->AdjacentTriangle(tri, id);
212  if (adjTri != -1 && a_flags[adjTri] == BE_UNVISITED)
213  {
214  found = true;
215  lastTri = tri + 1;
216  }
217  }
218  }
219  if (!found)
220  {
221  ++tri;
222  }
223  }
224 
225  // all have been flagged, we're done
226  if (!found)
227  {
228  done = true;
229  // AKZ - flag unvisited as out
230  for (size_t i = 0; i < a_flags.size(); ++i)
231  {
232  if (a_flags[i] == BE_UNVISITED)
233  {
234  a_flags[i] = BE_OUTSIDE;
235  }
236  }
237  }
238  else
239  {
240  int size = 0;
241  // push the first tri onto the stack
242  stack[size++] = tri;
243  for (int i = 0; i < size; i++)
244  {
245  int currTri = stack[i];
246  a_flags[currTri] |= BE_VISITED;
247  // find the unflagged adjacents and push onto stack
248  int id = 0;
249  do
250  {
251  int adjTri = m_tin->AdjacentTriangle(currTri, id);
252  if (adjTri != -1 && a_flags[adjTri] == BE_UNVISITED)
253  {
254  // push adjacent onto stack
255  stack[size++] = adjTri;
256  if (a_flags[currTri] & BE_INSIDE)
257  a_flags[adjTri] = BE_INSIDE;
258  else
259  a_flags[adjTri] = BE_OUTSIDE;
260  }
261  id = trIncrementIndex(id);
262  } while (id != 0);
263  }
264  }
265  }
266 
267 } // TrOuterTriangleDeleterImpl::MarkNeighbors
