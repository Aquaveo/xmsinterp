219 {
220  int ptIdx[3];
221  for (int i = 0; i < 3; ++i)
222  ptIdx[i] = a_ptIdxs[i];
223  if (ptIdx[0] < 0 || ptIdx[1] < 0 || ptIdx[2] < 0)
224  return false;
225  x1 = m_pts[ptIdx[0]].x;
226  p1 = m_pts[ptIdx[0]].y;
227  f1 = m_pts[ptIdx[0]].z;
228 
229  x2 = m_pts[ptIdx[2]].x;
230  y2 = m_pts[ptIdx[2]].y;
231  f2 = m_pts[ptIdx[2]].z;
232 
233  x3 = m_pts[ptIdx[1]].x;
234  y3 = m_pts[ptIdx[1]].y;
235  f3 = m_pts[ptIdx[1]].z;
236 
237  delta = x3 * p1 - x1 * y3 + x1 * y2 - x2 * p1 + x2 * y3 - x3 * y2;
238  a11 = (2.0 * p1 - y3 - y2) / delta;
239  a12 = (-2.0 * x1 + x2 + x3) / delta;
240  d1 = (x1 * (y3 + y2) - p1 * (x3 + x2)) / delta;
241  a21 = (-SQRT3 * (y3 - y2)) / delta;
242  a22 = (SQRT3 * (x3 - x2)) / delta;
243  d2 = 2 / SQRT3 + (SQRT3 * (x1 * (y3 - y2) - p1 * (x3 - x2))) / delta;
244 
245  d11 = 0.5 * (x3 - x2);
246  d12 = (1.0 / (2.0 * SQRT3)) * (2.0 * x1 - x2 - x3);
247  d21 = 0.5 * (y3 - y2);
248  d22 = (1.0 / (2.0 * SQRT3)) * (2.0 * p1 - y2 - y3);
249 
250  fx1 = fy1 = fx2 = fy2 = fx3 = fy3 = 0;
251  if (m_nodalFunc)
252  {
253  Pt3d grad;
254  m_nodalFunc->GradientFromPtIdx(ptIdx[0], grad);
255  fx1 = grad.x;
256  fy1 = grad.y;
257  m_nodalFunc->GradientFromPtIdx(ptIdx[2], grad);
258  fx2 = grad.x;
259  fy2 = grad.y;
260  m_nodalFunc->GradientFromPtIdx(ptIdx[1], grad);
261  fx3 = grad.x;
262  fy3 = grad.y;
263  }

266  un1 = d12 * fx1 + d22 * fy1;
267  ue2 = d11 * fx2 + d21 * fy2;
