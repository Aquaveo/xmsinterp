492 //------------------------------------------------------------------------------
493 void InterpIdwImpl::InterpWeights(const Pt3d& a_pt,
494  std::vector<int>& a_nPts,
495  std::vector<double>& a_w) const
496 {
497  std::vector<double> d2;
498  // first see if we are at this location
499  m_ptSearch->NearestPtsToPt(a_pt, 1, false, a_nPts);
500  // get the distance between our location and the nearest point
501  inDistanceSquared(a_pt, a_nPts, *m_pts, m_2d, d2);
502  if (m_nNearestPts == 1 || d2[0] < gmXyTol())
503  {
504  a_w.resize(1);
505  a_w[0] = 1;
506  return;
507  }
508  // see if we are doing all points
509  if (m_nNearestPts < 0)
510  {
511  a_nPts.assign(m_pts->size(), 0);
512  for (size_t i = 0; i < m_pts->size(); ++i)
513  a_nPts[i] = (int)i;
514  }
515  else
516  {
517  m_ptSearch->NearestPtsToPt(a_pt, m_nNearestPts, m_quadOctSearch, a_nPts);
518  }
519  // compute weights for each point
520  inDistanceSquared(a_pt, a_nPts, *m_pts, m_2d, d2);
521  inIdwWeights(d2, m_power, m_modifiedShepardWeights, a_w);
522 } // InterpIdwImpl::InterpWeights
