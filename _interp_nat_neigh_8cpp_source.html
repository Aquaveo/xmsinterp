296 void InterpNatNeighImpl::FillCenterVec()
297 {
298  if (m_pts.empty() || m_tris.empty())
299  return;
300 
301  xms::Pt3d pt, ptMin, ptMax;
302  ptMin = ptMax = m_pts[0];
303  for (size_t i = 1; i < m_pts.size(); ++i)
304  {
305  ptMin.x = Mmin(m_pts[i].x, ptMin.x);
306  ptMin.y = Mmin(m_pts[i].y, ptMin.y);
307  ptMax.x = Mmax(m_pts[i].x, ptMax.x);
308  ptMax.y = Mmax(m_pts[i].y, ptMax.y);
309  }
310  m_tol = gmXyDistance(ptMin, ptMax) / 1e9;
311  m_centers.assign(m_tris.size() / 3, xms::Pt3d());
312  for (size_t i = 0, cnt = 0; i < m_tris.size(); i += 3, ++cnt)

315  // calculate the center (x,y) and radius squared (z) of the circumcenter
316  // of the triangle
317  gmCircumcircleWithTol(&p0, &p1, &p2, &pt.x, &pt.y, &pt.z, m_tol);

362 {
363  a_tris.resize(0);

365  return;

367  if (XM_NONE == tIdx)
368  return;
369 
370  a_tris.push_back(tIdx);
371  std::vector<int> visited(m_tris.size(), 0);
372  visited[tIdx] = 1;
373  for (size_t j = 0; j < a_tris.size(); ++j)
374  {
375  tIdx = a_tris[j];
376  std::vector<int> vals;
377  NeighTriFromTriIdx(tIdx, vals);
378  xms::Pt3d center;
379  double rSquared, distSquared;
380  std::set<int> ptIdxs;
381  // now check to see if the point is inside the circumcircle
382  for (size_t i = 0; i < vals.size(); ++i)
383  {
384  size_t tri = vals[i];
385  if (visited[tri])
386  continue;
387  visited[tri] = 1;

459 //------------------------------------------------------------------------------
460 void InterpNatNeighImpl::HaleNnVisitNeighbors(int a_tIdx,
461  const xms::Pt3d& a_pt,
462  std::vector<nnOuterEdgeStruct>& a_outerEdges,
463  std::map<int, double>& a_weights)
464 {
465  std::set<std::pair<int, int>> processedEdges;
466  std::vector<int> tris(1, a_tIdx);
467  for (size_t t = 0; t < tris.size(); ++t)
468  { // visit adjacent triangles
469  std::pair<int, int> edge;
470  int tIdx = tris[t];
471  for (int i = 0; i < 3; ++i)
472  {
473  int ptIdx = m_tris[tIdx + i];
474  int nextTriPtIdx = i == 2 ? 0 : i + 1;
475  int ptIdx1 = m_tris[tIdx + nextTriPtIdx];
476  edge = iMakePair(ptIdx, ptIdx1);
477  // already processed this edge
478  if (processedEdges.find(edge) != processedEdges.end())
479  continue;
480 
481  // see if there is an adjacent triangle and if it is a natural neighbor
482  int adjTri = AdjacentTriangle(edge, tIdx);
483  if (XM_NONE == adjTri || !PtInTriCircumCircle(a_pt, adjTri))
484  { // no adjtri or not a natural neighbor. This is an outer edge.
485  HaleNnAddOuterEdge(a_outerEdges, tIdx, ptIdx, ptIdx1, a_pt);
486  continue;
487  }
488 
489  // adjacent triangle is a natural neighbor. Calculate area contribution.
490  // cross the circumcenters of the 2 triangles.
491  const xms::Pt3d &tCC(m_centers[tIdx / 3]), adjCC(m_centers[adjTri / 3]);
492  double cross = (tCC.x * adjCC.y) - (tCC.y * adjCC.x);
493  a_weights[ptIdx] -= cross;
494  a_weights[ptIdx1] += cross;
495 
496  tris.push_back(adjTri);
497  // add to processed edges
498  processedEdges.insert(edge);
499  }
500  }
501 } // InterpNatNeighImpl::HaleNnVisitNeighbors

578 void InterpNatNeighImpl::HaleNnOuterEdgesToWeights(std::vector<nnOuterEdgeStruct>& a_outerEdges,
579  std::map<int, double>& a_weights)
580 {
581  for (size_t i = 0; i < a_outerEdges.size(); ++i)
582  { // get the current edge and the next edge
583  nnOuterEdgeStruct& curr(a_outerEdges[i]);
584  size_t nextIdx = i + 1 == a_outerEdges.size() ? 0 : i + 1;
585  nnOuterEdgeStruct& next(a_outerEdges[nextIdx]);
586 
587  int ptIdx = curr.m_edge[1];
588  // cross the original tri cc with the tri cc formed by curr edge and the
589  // interpolation point
590  {
591  xms::Pt3d &o1(m_centers[curr.m_triIdx / 3]), &c1(curr.m_cc);
592  double cross = (o1.x * c1.y) - (o1.y * c1.x);
593  a_weights[ptIdx] += cross;
594  a_weights[curr.m_edge[0]] -= cross;
595  }
596 
597  // cross tri cc formed by curr edge and interp pt with tri cc formed by
598  // next edge and interp pt
599  {
600  xms::Pt3d &o1(curr.m_cc), &c1(next.m_cc);
601  double cross = (o1.x * c1.y) - (o1.y * c1.x);
602  a_weights[ptIdx] += cross;
603  }
604  }
