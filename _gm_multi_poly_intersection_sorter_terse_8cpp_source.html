112 void GmMultiPolyIntersectionSorterTerse::RemoveDuplicateEdges() {
113  XM_ENSURE_TRUE_VOID_NO_ASSERT(m_d->m_ixs.size() > 1);
114 
115  VecInt tChange;
116  FindWhereTValuesChange(tChange);
117 
118  VecInt toRemove;
119 
120  // Loop through tChanges
121  for (size_t i = 0; i < tChange.size() - 1; ++i) {
122  if (tChange[i + 1] - tChange[i] > 1) {
123  VecInt inNext;
124  // Get intersections in this tvalue group that also appear in next group.
125  // These are the first points of the edges.
126  FindPreviousNextNeither(tChange, (int)i, nullptr, &inNext, nullptr);
127 
128  if (inNext.size() > 1) {
129  // Get intersections in next tvalue group that match those in current.
130  // These are the last points of the edges.
131  VecInt inPrev;
132  FindPreviousNextNeither(tChange, (int)i + 1, &inPrev, nullptr, nullptr);
133  // XM_ASSERT(inPrev.size() == inNext.size());
134  // Not sure why the assert above was put in. Even when it asserts,
135  // we still seem to get good results in the end.
136 
137  // Keep only the polygons that are on the right of the line
138  bool found = false; // Should be just 2 so stop if we find one
139  for (size_t j = 0; j < inNext.size() && !found; ++j) {
140  // Find the other end of this edge
141  int edgeBeg = inNext[j]; // index into m_d->m_ixs of beginning of edge
142  int edgeEnd = XM_NONE; // index into m_d->m_ixs of end of edge
143  for (size_t k = 0; k < inPrev.size() && edgeEnd == XM_NONE; ++k) {
144  if (m_d->m_ixs[inPrev[k]].m_i == m_d->m_ixs[edgeBeg].m_i)
145  edgeEnd = inPrev[k];
146  }
147  XM_ASSERT(edgeEnd != XM_NONE);
148 
149  int polyIdx = m_d->m_ixs[edgeBeg].m_i - 1;
150  // Create the poly as a vector of Pt3ds for gmComputeCentroid
151  VecPt3d poly;
152  for (size_t k = 0; k < m_d->m_polys[polyIdx].size(); ++k) {
153  poly.push_back(m_d->m_points[m_d->m_polys[polyIdx][k]]);
154  }
155  Pt3d centroid = gmComputeCentroid(poly);
156  Turn_enum turn = gmTurn(m_d->m_ixs[edgeBeg].m_pt,
157  m_d->m_ixs[edgeEnd].m_pt, centroid);
158  if (turn != TURN_RIGHT) {
159  toRemove.push_back(edgeBeg);
160  toRemove.push_back(edgeEnd);
161  found = true;
162  }
163  }
164  }
165  }
166  }

168  // Remove them, working backwards
169  std::sort(toRemove.begin(), toRemove.end());
170  for (size_t i = toRemove.size(); i-- > 0;) {
171  m_d->m_ixs.erase(m_d->m_ixs.begin() + toRemove[i]);

178  XM_ENSURE_TRUE_VOID_NO_ASSERT(m_d->m_ixs.size() > 1);

260  VecInt inPrev, inNext, inNeither;

265  if (inPrev.empty() && inNeither.empty())
266  inNeither = inNext;
267

271  AddToPolyIdsAndTValues(m_d->m_ixs[inNeither[j]], polyids, tvalues,
272  a_pts);
273  }
274  } else {
275  // Take the ones that were found in the previous group, that were
276  // not added previously
277  for (size_t j = 0; j < inPrev.size(); ++j) {

289  if (setIdsPrev.find(m_d->m_ixs[inNext[j]].m_i) ==

293  a_pts);
294  }
295  }
296  }

302  a_pts);
303  }
304

363 void GmMultiPolyIntersectionSorterTerse::FixArrays(VecInt &polyids,
364  VecDbl &tvalues,
365  VecPt3d &a_pts) const {
366  XM_ENSURE_TRUE_VOID_NO_ASSERT(!polyids.empty());
367 
368  // Make sure we have at least two values - an entrance and exit
369  if (polyids.size() == 1) {
370  // Only one intersection. Add an exit that is the same
371  polyids.push_back(polyids.back());
372  tvalues.push_back(tvalues.back());
373  a_pts.push_back(a_pts.back());
374  }
375 
376  // If crossing a vertex, make sure we have an exit
377  if (m_d->m_polys2.empty() && polyids.back() != polyids[polyids.size() - 2]) {
378  polyids.push_back(polyids.back());
379  tvalues.push_back(tvalues.back());
380  a_pts.push_back(a_pts.back());
381  }
382 
383  // Make sure we always end with -1
384  if (polyids.size() > 1) {
385  polyids.back() = -1;
386  }
387 
388  // If we leave and reenter, set exit to -1
389  for (size_t i = 1; i < polyids.size(); ++i) {
390  if (polyids[i] == polyids[i - 1]) {
391  polyids[i] = -1;
392  }
393  }

424 void GmMultiPolyIntersectionSorterTerse::FindPreviousNextNeither(
425  const VecInt &tChange, const int idx, VecInt *inPrev, VecInt *inNext,
426  VecInt *inNeither) const {
427  if (inPrev)
428  inPrev->clear();
429  if (inNext)
430  inNext->clear();
431  if (inNeither)
432  inNeither->clear();
433  for (int j = tChange[idx]; j < tChange[idx + 1]; ++j) {
434  // Corners at the start of the list of intersections appear the same
435  // as edges so we can't remove them because we don't have a previous
436  // set of tvalues to look at.
437  bool foundPrev = false;
438  if (idx > 0) {
439  for (int k = tChange[idx - 1]; k < tChange[idx] && !foundPrev; ++k) {
440  if (m_d->m_ixs[j].m_i == m_d->m_ixs[k].m_i) {
441  foundPrev = true;
442  }
443  }
444  }
445 
446  // Corners at the end of the list of intersections appear the same
447  // as edges so we can't remove them because we don't have a inNext set
448  // of tvalues to look at.
449  bool foundNext = false;
450  if (!foundPrev && idx + 2 < (int)tChange.size()) {
451  for (int k = tChange[idx + 1]; k < (int)tChange[idx + 2] && !foundNext;
452  ++k) {
453  if (m_d->m_ixs[j].m_i == m_d->m_ixs[k].m_i) {
454  foundNext = true;
455  }
456  }
457  }
458 
459  if (foundPrev) {
460  if (inPrev)
461  inPrev->push_back(j);
462  } else if (foundNext) {
463  if (inNext)
464  inNext->push_back(j);
465  } else {
466  if (inNeither)
467  inNeither->push_back(j);
468  }
469  }
