123 void GmMultiPolyIntersectionSorterTerse::RemoveDuplicateEdges()
124 {
125  XM_ENSURE_TRUE_VOID_NO_ASSERT(m_d->m_ixs.size() > 1);
126 
127  VecInt tChange;
128  FindWhereTValuesChange(tChange);
129 
130  VecInt toRemove;
131 
132  // Loop through tChanges
133  for (size_t i = 0; i < tChange.size() - 1; ++i)
134  {
135  if (tChange[i + 1] - tChange[i] > 1)
136  {
137  VecInt inNext;
138  // Get intersections in this tvalue group that also appear in next group.
139  // These are the first points of the edges.
140  FindPreviousNextNeither(tChange, (int)i, nullptr, &inNext, nullptr);
141 
142  if (inNext.size() > 1)
143  {
144  // Get intersections in next tvalue group that match those in current.
145  // These are the last points of the edges.
146  VecInt inPrev;
147  FindPreviousNextNeither(tChange, (int)i + 1, &inPrev, nullptr, nullptr);
148  // XM_ASSERT(inPrev.size() == inNext.size());
149  // Not sure why the assert above was put in. Even when it asserts,
150  // we still seem to get good results in the end.
151 
152  // Keep only the polygons that are on the right of the line
153  bool found = false; // Should be just 2 so stop if we find one
154  for (size_t j = 0; j < inNext.size() && !found; ++j)
155  {
156  // Find the other end of this edge
157  int edgeBeg = inNext[j]; // index into m_d->m_ixs of beginning of edge
158  int edgeEnd = XM_NONE; // index into m_d->m_ixs of end of edge
159  for (size_t k = 0; k < inPrev.size() && edgeEnd == XM_NONE; ++k)
160  {
161  if (m_d->m_ixs[inPrev[k]].m_i == m_d->m_ixs[edgeBeg].m_i)
162  edgeEnd = inPrev[k];
163  }
164  XM_ASSERT(edgeEnd != XM_NONE);
165 
166  int polyIdx = m_d->m_ixs[edgeBeg].m_i - 1;
167  // Create the poly as a vector of Pt3ds for gmComputeCentroid
168  VecPt3d poly;
169  for (size_t k = 0; k < m_d->m_polys[polyIdx].size(); ++k)
170  {
171  poly.push_back(m_d->m_points[m_d->m_polys[polyIdx][k]]);
172  }
173  Pt3d centroid = gmComputeCentroid(poly);
174  Turn_enum turn = gmTurn(m_d->m_ixs[edgeBeg].m_pt, m_d->m_ixs[edgeEnd].m_pt, centroid);
175  if (turn != TURN_RIGHT)

178  toRemove.push_back(edgeEnd);
179  found = true;
180  }
181  }
182  }
183  }
184  }

416 {
