171 void TrTriangulatorPoints::ReceiveTriangle(int a_id1, int a_id2, int a_id3)
172 {
173  // make sure the triangle has positive area
174  double area = trArea(m_pts[a_id1 - 1], m_pts[a_id2 - 1], m_pts[a_id3 - 1]);
175  if (area < m_areaTol)
176  {
177  return;
178  }
179 
180  m_tris.push_back(a_id1 - 1);
181  m_tris.push_back(a_id2 - 1);
182  m_tris.push_back(a_id3 - 1);
183 
184  if (m_trisAdjToPts)
185  {
186  int triIdx = ((int)m_tris.size() / 3) - 1;
187  (*m_trisAdjToPts)[a_id1 - 1].push_back(triIdx);
188  (*m_trisAdjToPts)[a_id2 - 1].push_back(triIdx);
189  (*m_trisAdjToPts)[a_id3 - 1].push_back(triIdx);
190  }
191 } // TrTriangulatorPoints::ReceiveTriangle

195 void TrTriangulatorPoints::UpdateAreaTolerance()
196 {
197  if (m_pts.empty())
198  return;
199 
200  Pt3d ptMin(m_pts[0]), ptMax(m_pts[0]);
201  for (const auto& p : m_pts)
202  gmAddToExtents(p, ptMin, ptMax);
203  m_areaTol = gmComputeXyTol(ptMin, ptMax);
204  m_areaTol *= m_areaTol;
205  m_areaTol *= 100;
206 } // TrTriangulatorPoints::UpdateAreaTolerance
