306 void inIdwWeights(const std::vector<double>& a_distSquare,
307  double a_power,
308  bool a_modifiedShepardWeights,
309  std::vector<double>& a_w)
310 {
311  std::vector<double> d2(a_distSquare);
312  double distSum(0), dExp(a_power / 2), r, d;
313  a_w.resize(0);
314  a_w.assign(d2.size(), 0);
315  r = sqrt(*std::max_element(d2.begin(), d2.end()));
316  for (size_t i = 0; i < d2.size(); ++i)
317  { // this is basically at the same location as our point
318  if (d2[i] < 1e-20)
319  d2[i] = 1e10;
320  }
321  for (size_t i = 0; i < d2.size(); ++i)
322  {
323  if (a_modifiedShepardWeights)
324  { // Franke, R., and Neilson, G. 1980. Smooth interpolation of large sets
325  // of scattered data. International Journal of Numerical Methods
326  // in Engineering 15, 1691-1704.
327 
328  // r is the distance to the furthest point, d is the distance to the
329  // current point
330  d = sqrt(d2[i]);
331  d2[i] = xms::sqr((r - d) / (r * d));
332  }
333  else
334  { // classic weighting function
335  if (dExp == 1)
336  { // 1 over distance squared
337  d2[i] = 1 / d2[i];
338  }
339  else
340  { // if the m_power variable is not 2 then we have to use the "pow" function
341  d2[i] = 1 / pow(d2[i], dExp);
342  }
343  }
344  distSum += d2[i];
345  }
346  if (0.0 == distSum)
347  {
348  d2 = a_distSquare;
349  for (size_t i = 0; i < d2.size(); ++i)
350  distSum += d2[i];
351  }
352  for (size_t i = 0; i < d2.size(); ++i)
353  {
354  a_w[i] = d2[i] / distSum;
355  }
356 } // inIdwWeights

364 bool inAllScalarsEqual(const std::vector<float>& a_scalars, const DynBitset& a_act)
365 {
366  bool allScalarsSame(true);
367  if (a_scalars.empty())
368  return allScalarsSame;
369 
370  if (a_act.empty() || a_act.size() != a_scalars.size())
371  {
372  float f = a_scalars[0];
373  for (size_t i = 1; allScalarsSame && i < a_scalars.size(); ++i)
374  {
375  if (f != a_scalars[i])
376  allScalarsSame = false;
377  }
378  return allScalarsSame;
379  }
380 
381  // get first active value
382  size_t idx(0);
383  if (!a_act[0])
384  {
385  for (size_t i = 1; idx == 0 && i < a_act.size(); ++i)
386  {
387  if (a_act[i])
388  idx = i;
389  }
390  }
391 
392  float f = a_scalars[idx];
393  for (size_t i = idx + 1; allScalarsSame && i < a_scalars.size(); ++i)
394  {
395  if (a_act[i] && f != a_scalars[i])
396  allScalarsSame = false;
397  }
398  return allScalarsSame;
