242 float InterpLinearImpl::InterpToPt(const Pt3d& a_pt)
243 {
244  // return m_triSearch->InterpToPt(a_pt);
245  float rval((float)m_extrap);
246  VecInt idxs(3, 0);
247  VecDbl wts(3, 0);
248  if (m_triSearch->InterpWeights(a_pt, idxs, wts))
249  {
250  float* s = &(*m_scalar)[0];
251  rval = (float)((s[idxs[0]] * wts[0]) + (s[idxs[1]] * wts[1]) + (s[idxs[2]] * wts[2]));
252  if (m_cloughTocher || m_natNeigh)
253  {
254  if (EQ_TOL(wts[0], 1, FLT_EPSILON) || EQ_TOL(wts[1], 1, FLT_EPSILON) ||
255  EQ_TOL(wts[2], 1, FLT_EPSILON))
256  {
257  }
258  else
259  {
260  if (m_cloughTocher)
261  {
262  rval = CloughTocherInterp(idxs, a_pt);
263  }
264  else
265  rval = NatNeighInterp(a_pt);
266  }
267  }
268  }
269  if (m_trunc)
270  {
271  if (rval < m_truncMin)
272  rval = (float)m_truncMin;
273  if (rval > m_truncMax)
274  rval = (float)m_truncMax;
275  }
276  return rval;
277 } // InterpLinearImpl::InterpToPt
