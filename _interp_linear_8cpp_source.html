321 float InterpLinearImpl::InterpToPt(const Pt3d& a_pt)
322 {
323  // return m_triSearch->InterpToPt(a_pt);
324  float rval((float)m_extrap);
325  VecInt idxs(3, 0);
326  VecDbl wts(3, 0);
327  if (m_triSearch->InterpWeights(a_pt, idxs, wts))
328  {
329  float* s = &(*m_scalar)[0];
330  rval = (float)((s[idxs[0]] * wts[0]) + (s[idxs[1]] * wts[1]) + (s[idxs[2]] * wts[2]));
331  if (m_cloughTocher || m_natNeigh)
332  {
333  if (EQ_TOL(wts[0], 1, FLT_EPSILON) || EQ_TOL(wts[1], 1, FLT_EPSILON) ||
334  EQ_TOL(wts[2], 1, FLT_EPSILON))
335  {
336  }
337  else
338  {
339  if (m_cloughTocher)
340  {
341  rval = CloughTocherInterp(idxs, a_pt);
342  }
343  else
344  rval = NatNeighInterp(a_pt);
345  }
346  }
347  }
348  if (m_trunc)
349  {
350  if (rval < m_truncMin)
351  rval = (float)m_truncMin;
352  if (rval > m_truncMax)
353  rval = (float)m_truncMax;
354  }
355  return rval;
356 } // InterpLinearImpl::InterpToPt
