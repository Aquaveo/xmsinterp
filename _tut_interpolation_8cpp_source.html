169 void TutInterpolationIntermediateTests::test_Example_Anisotropic()
170 {
171  // Create the centerline points. The segments of this polyline will be mapped
172  // onto the x-axis. The total length will be preserved.
173  xms::VecPt3d centerline = {{0, 0, 0}, {20, 0, 0}, {40, 20, 0}, {60, 20, 0}};
174 
175  // Create the points used to interpolate. These points are typically taken
176  // from cross sections across the center line. These points will be
177  // transformed so that the x-coordinate corresponds to the intersection of a
178  // line through the point and perpendicular to a segment of the centerline.
179  // The y coordinate will be the distance above or below that intersection.
180  // Note that a single point may project onto several segments, generating
181  // multiple transformed points.
182  xms::VecPt3d interpolationPoints = {
183  // cross-section 1
184  {0, 10, 100},
185  {0, -10, 100},
186  // cross-section 2
187  {10, 10, 90},
188  {10, 5, 60},
189  {10, 0, 50},
190  {10, -5, 70},
191  {10, -10, 90},
192  // cross-section 3
193  {20, 20, 80},
194  {30, 10, 40},
195  {40, 0, 80},
196  // cross-section 4
197  {30, 40, 70},
198  {35, 30, 50},
199  {40, 20, 30},
200  {45, 10, 70},
201  // cross-section 5
202  {60, 30, 50},
203  {60, 10, 50},
204  };
205 
206  // create the interpolator class
207  BSHP<xms::InterpAnisotropic> interpolator = xms::InterpAnisotropic::New();
208 
209  // Set the centerline and interpolation points used by the interpolator.
210  // Any point may project onto several segments of the centerline. You can use
211  // all such transformed points or set pickClosest to true to use only the
212  // one nearest the centerline.
213  bool pickClosest = false;
214  interpolator->SetPoints(centerline, interpolationPoints, pickClosest);
215 
216  // After the points are transformed, an inverse distance weighted
217  // interpolation is done (using all of the interpolation points) after scaling
218  // the points in the x direction. The default x scale factor is 1. Set to a
219  // value less than 1 to compress the x values (thus giving them more weight
220  // than y) or to a value greater than 1 (to have the opposite effect).
221  interpolator->SetXScale(0.5);
222 
223  // Set the exponent of the inverse distance weight to a value over 1 to dilute
224  // the influence of interpolation points far from the point in question.
225  interpolator->SetPower(3);
226 
227  // You can access the actual transformed interpolation points as shown below.
228  // This isn't necessary for the interpolation, but is useful if you want
229  // plot and visualize the transformation.
230  xms::VecPt3d snPoints;
231  interpolator->GetInterpolationPts(snPoints);
232  // clang-format off
233  xms::VecPt3d expectedSnPoints = {
234  // cross-section 1
235  {0, 10, 100}, // 0
236  {0, -10, 100}, // 1
237  // cross-section 2
238  {5, 10, 90}, // 2
239  {10, 14.142, 90}, // 2
240  {5, 5, 60}, // 3
241  {5, 0, 50}, // 4
242  {5, -5, 70}, // 5
243  {5, -10, 90}, // 6
244  // cross-section 3
245  {10, 20, 80}, // 7
246  {17.071, 14.142, 80}, // 7
247  {17.071, 0, 40}, // 8
248  {17.071, -14.142, 80}, // 9
249  {24.142, -20, 80}, // 9
250  // cross-section 4
251  {24.142, 22.361, 70}, // 10
252  {24.142, 11.180, 50}, // 11
253  {24.142, 0, 30}, // 12
254  {24.142, 0, 30}, // 12
255  {22.374, -10.607, 70}, // 13
256  {26.642, -10, 70}, // 13
257  // cross-section 5
258  {34.142, 10, 50}, // 14
259  {34.142, -10, 50} // 15
260  };
261  TS_ASSERT_DELTA_VECPT3D(expectedSnPoints, snPoints, 0.01);
262  // interpolate to a location and verify the value.
263  xms::Pt3d loc(20, 5, 0);
264  float val = interpolator->InterpToPt(loc);
265  float base(59.5313f);
266  TS_ASSERT_DELTA(base, val, 0.001);
267 
268  // interpolate to several locations and verify the values. Note that the
269  // last two points are beyond the range of the centerline; hence, they
270  // generate no data (XM_NODATA) interpolation values.
271  xms::VecPt3d interpToPoints = {
272  {5, 5, 0}, {5, 0, 0}, {5, -5, 0},
273  {20, 5, 0},
274  {20, 0, 0},
275  {20, -5, 0}, {10, 20, 0}, {30, -5, 0}, {30, 30, 0},
276  {35, 20, 0}, {45, 25, 0}, {45, 15, 0},
277  {65, 20, 0}, {-5, 0, 0}};
278 
279  // The transformed values of interpolated to points can be retrieved by
280  // using GetTransformedPts. The transformed points could be used for
281  // plotting or using another interpolation method.
282  xms::VecPt3d snInterpToPoints;
283  interpolator->GetTransformedPts(interpToPoints, false, snInterpToPoints);
284  xms::VecPt3d expectedSnInterpToPoints = {
285  {2.5, 5, 0}, {2.5, 0, 0}, {2.5, -5, 0}, // points 0, 1, 2
286  {10, 5, 0}, {11.768, 3.536, 0}, // point 3 is transformed to 2 stations
287  {10, 0, 0}, {10, 0, 0}, // point 4 is transformed to segment 0 { param 1 } and segment 1 { param 0 }
288  {10, -5, 0}, {5, 20, 0}, {13.536, 21.213, 0}, {11.768, -10.607, 0}, // points 5, 6, 7
289  {24.142, 14.142, 0}, {22.374, 3.536, 0}, {26.642, 5, 0}, // points 8, 9, 10
290  {24.142, -7.071, 0}, {26.642, -5, 0}}; // point 11 is transformed to 2 stations
291  // clang-format on
292  TS_ASSERT_DELTA_VECPT3D(expectedSnInterpToPoints, snInterpToPoints, 0.001);
293 
294  xms::VecFlt interpValues;
295  interpolator->InterpToPts(interpToPoints, interpValues);
296  xms::VecFlt expectedInterpValues = {64.614f, 54.390f, 71.970f, 59.531f, 58.468f,
297  68.0917f, 81.688f, 76.032f, 53.130f, 35.413f,
298  40.457f, 50.839f, (float)XM_NODATA, (float)XM_NODATA};
299  TS_ASSERT_DELTA_VEC(expectedInterpValues, interpValues, 0.01);
300 
301 } // TutInterpolationIntermediateTests::test_Example_Anisotropic
